To enforce rules I ceate services.
For example for the rule “a recipe must have at least one ingredient”, I create a RecipeService in the Domain/Services layer

Example code:
    public class RecipeService
    {
        private readonly IRecipeRepository _recipeRepository;

        public RecipeService(IRecipeRepository recipeRepository)
        {
            _recipeRepository = recipeRepository;
        }

        public async Task AddRecipeAsync(Recipe recipe)
        {
            if (string.IsNullOrWhiteSpace(recipe.Name))
                throw new ArgumentException("Recipe name cannot be empty.");

            if (recipe.Steps.Count < 1)
                throw new InvalidOperationException("Recipe must have at least one step.");

            if (recipe.RecipeIngredients.Count < 1)
                throw new InvalidOperationException("Recipe must have at least one ingredient.");

            if (recipe.RecipeCategories.Count < 1)
                throw new InvalidOperationException("Recipe must have at least one category.");

            await _recipeRepository.AddAsync(recipe);
            await _recipeRepository.SaveChangesAsync();
        }
    }
